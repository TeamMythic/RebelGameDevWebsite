Grab Component Methods

Below is an example how to call a static method from the library. 

                            
    //You need to add this to grab the RGD static class library:
    using static RebelGameDevs.Utils.World.RGD_GrabComponentMethods;    

    public class ShootRayFromCamScriptExample : MonoBehaviour     
    {
            //Players cam:
            [SerializeField] private cam;
            //Distance of ray:
            [SerializeField] private  float distance =  4.0f;

            private void Update ()
            {
                if(RGD_TryGrabRayCasterFromCam(cam, distance, out RaycastHit hitResult,  out Collider col))     
                {
                        //If the raycast hits an gameobject that has a Collider:
                        Debug.Log("Shooting Raycast");
                    }
            }
    }
